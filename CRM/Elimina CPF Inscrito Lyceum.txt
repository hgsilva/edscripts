delete LY_OPCOES_PROC_SELETIVO
from   LY_OPCOES_PROC_SELETIVO a
where  CANDIDATO in (
                                               select candidato
                                               from   LY_CANDIDATO b
                                               where  CPF = @cpf
                                               and   a.CONCURSO = b.concurso
                                        )

from	LY_BOLETOS_CANCELADOS ab 
join LY_CANDIDATO ac on ab.CANDIDATO = ac.CANDIDATO and ab.CONCURSO = ac.CONCURSO
 where  CPF = @cpf
